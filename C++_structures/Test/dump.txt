		bool binarySearch(List<int>* MyList, int i){
			int inferior = 0, superior = (MyList->counterList()-1);
			while (inferior <= superior){
				int middle = ((inferior-superior)/2);
				if (MyList->get(middle) < i){
					inferior = middle -1;
				}else if(MyList->get(middle) > i){
					superior = middle +1;
				}else {
					return true;
				}
			}
			return false;
		}
